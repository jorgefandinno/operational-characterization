    \(
    P
    =
    \left\{
      \begin{array}{lll}
        a \leftarrow                \quad &
        c \leftarrow a, \neg d      \quad &
        e \leftarrow b, \neg f
        \\
        b \leftarrow \neg a         \quad &
        d \leftarrow \neg c, \neg e \quad &
        e \leftarrow e
      \end{array}
    \right\}
    \)
    \bigskip
    \bigskip
  \item \structure{Spotlight} \quad
    \(
    {\mathbf{\Omega}}_{P}\langle\{c\},\emptyset \rangle=\langle\{a\},\{d,f\}\rangle
    \)
  \end{itemize}
\end{frame}
